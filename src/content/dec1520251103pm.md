Today not so bad, but i need to review the fail test cases to find the answer, to solve this problem we need to use flags, to control the response of "YES" or "NO" and to count the number of occurrences of each letter in the password is 2, for that we need three flags and trick is that it's possible found the password if the first character is equals to the second character or the x word and the second character(password) is equals to the first letter of x word and that is all, this is an easily recognizable pattern.


> problem: <https://codeforces.com/problemset/problem/868/A>


``` java
import java.io.*;
import java.util.*;

public class BarkToUnlock {
    public static void main(String[] args) {
        Kattio io = new Kattio();

        String pass = io.next();
        int n = io.nextInt();
        int countA = 0, countB = 0;

        char x = pass.charAt(0);
        char x1 = pass.charAt(1);

        boolean found = false;

        for(int i=0; i<n; i++){
            String word = io.next();

            if(!found) {
                char a = word.charAt(0);
                char b = word.charAt(1);
                if(x == b && x1 == a || x == a && x1 == b) {
                    found = true;
                }else if (x == b && x1 != a && countA == 0) {
                    countA++;
                }else if (x1 == a && x != b && countB == 0) {
                    countB++;
                }

                if(countA + countB == 2){
                    found = true;
                }
            }
        }
        if(found) io.println("YES");
        else io.println("NO");

        io.close();
    }

    static class Kattio extends PrintWriter {
        private BufferedReader r;
        private StringTokenizer st;
        // standard input
        public Kattio() { this(System.in, System.out); }
        public Kattio(InputStream i, OutputStream o) {
            super(o);
            r = new BufferedReader(new InputStreamReader(i));
        }
        // USACO-style file input
        public Kattio(String problemName) throws IOException {
            super(problemName + ".out");
            r = new BufferedReader(new FileReader(problemName + ".in"));
        }
        // returns null if no more input
        public String next() {
            try {
                while (st == null || !st.hasMoreTokens())
                    st = new StringTokenizer(r.readLine());
                return st.nextToken();
            } catch (Exception e) {}
            return null;
        }
        public int nextInt() { return Integer.parseInt(next()); }
        public double nextDouble() { return Double.parseDouble(next()); }
        public long nextLong() { return Long.parseLong(next()); }
    }

}
```
