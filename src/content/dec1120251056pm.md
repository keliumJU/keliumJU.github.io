Today was a easy day i think, but that don't care, the problem was a classic of the math the weird problem of CSES https://cses.fi/problemset/task/1068 only follow the instructions when a number is even divide by 2 otherwise multiply by 3 and plus 1, so that is all, the main learning of this day was the input and output format, for the java lang we use the classic system with Scanner and System.out - System.in, but also the stream approach with InputStream and OutputStream, i only use the Kattio template without deep, follow i share the solution with this approach.


``` java
import java.io.*;
import java.util.*;

/**
 * Simple yet moderately fast I/O routines.
 * Some notes:
 *
 * - When done, you should always do io.close() or io.flush() on the
 *   Kattio-instance, otherwise, you may lose output.
 *
 * - The nextInt(), nextDouble(), and nextLong() methods will throw an
 *   exception if there is no more data in the input.
 *
 * @author: Kattis
 */

class Kattio extends PrintWriter {
    private BufferedReader r;
    private StringTokenizer st;
    // standard input
    public Kattio() { this(System.in, System.out); }
    public Kattio(InputStream i, OutputStream o) {
        super(o);
        r = new BufferedReader(new InputStreamReader(i));
    }
    // USACO-style file input
    public Kattio(String problemName) throws IOException {
        super(problemName + ".out");
        r = new BufferedReader(new FileReader(problemName + ".in"));
    }
    // returns null if no more input
    public String next() {
        try {
            while (st == null || !st.hasMoreTokens())
                st = new StringTokenizer(r.readLine());
            return st.nextToken();
        } catch (Exception e) {}
        return null;
    }
    public int nextInt() { return Integer.parseInt(next()); }
    public double nextDouble() { return Double.parseDouble(next()); }
    public long nextLong() { return Long.parseLong(next()); }
}

public class WeirAlgo {
    public static void main(String args[]) {
        Kattio io = new Kattio();

        long n = io.nextLong();

        while(n > 1) {
            io.print(n + " ");
            if(n%2==0) {
                n/=2;
            }else{
                n=n*3+1;
            }
        }

        io.print("1");

        io.close();
    }
}
```
